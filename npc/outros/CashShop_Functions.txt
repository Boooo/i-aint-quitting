// Cartão Kafra
//============================================================ 
// - Open player's storage.
// - No arguments.
function	script	F_CashStore	{

	mes "[Funcionária Kafra]";
	mes "Bem-vindo à Corporação Kafra.";

	return;
}


// Asa de Mosca Gigante
//============================================================ 
// - Teleporta o Líder do Grupo e seus integrantes para algum lugar aleartório no Mapa.
function	script	F_CashPartyCall	{
	warp "Random",0,0;
	if(getpartyleader(getcharid(1),2) == getcharid(0)) {
		getmapxy .@mapl$, .@xl, .@yl, 0;
		warpparty .@mapl$, .@xl, .@yl, getcharid(1), .@mapl$;
	}
	return;
}

// Neuralizer
//============================================================ 
// - Reseta os Pontos de Habilidade. Mas para isto você não pode estar com items ou Carrinho/Falcão/Peco-Peco.
function	script	F_CashReset	{
	if (Class != Job_Novice) {
		if (Weight < 1) { 
			if (checkriding() || checkfalcon() || checkcart()) return;
			else {
				resetskill;
				sc_end SC_ALL;
			}
		}
	}
	return;
}

			case 1: warp "rachel",0,0; end;
